      int safetoplace(int x[],int k)
      { int i;



      for(i=1;i<k;i++)



      {
      if(x[i]==x[k]||i-x[i]==k-x[k]||i+x[i]==k+x[k])
      {
      return 0; /* False*/
      }



      }
